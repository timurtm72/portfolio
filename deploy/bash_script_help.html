📋 Основы Bash скриптов

Создание первого скрипта

📖 Базовая структура скрипта: 
#!/bin/bash
# Это комментарий
# Первая строка (shebang) указывает интерпретатор

echo "Hello, World!"
echo "Мой первый bash скрипт"

Создание и запуск скрипта

nano myscript.sh
Создаём файл скрипта
chmod +x myscript.sh
Делаем скрипт исполняемым
./myscript.sh
Запускаем скрипт
💡 Способы запуска скрипта:
  • ./script.sh - запуск как исполняемый файл (требует chmod +x)
  • bash script.sh - запуск через bash (права не нужны)
  • sh script.sh - запуск через sh
  • source script.sh или . script.sh - выполнение в текущей оболочке

Shebang варианты

Shebang Описание
#!/bin/bash Использовать bash (рекомендуется)
#!/bin/sh Использовать sh (более совместимо, меньше функций)
#!/usr/bin/env bash Найти bash в PATH (более переносимо)
#!/bin/zsh Использовать zsh

Отладка скриптов

bash -x script.sh
Запуск с выводом отладочной информации
📖 Отладка внутри скрипта: 
#!/bin/bash

# Включить отладку
set -x

echo "Эта строка будет показана с отладкой"

# Выключить отладку
set +x

echo "Эта строка без отладки"

# Остановить при ошибке
set -e

# Показывать выполняемые команды
set -v
💡 Полезные опции set:
  • set -e - остановить скрипт при любой ошибке
  • set -u - ошибка при использовании неопределённых переменных
  • set -x - показывать выполняемые команды
  • set -o pipefail - ошибка если любая команда в pipeline упала
  • set -euxo pipefail - комбинация всех опций для строгого режима

📦 Переменные

Объявление и использование переменных

📖 Работа с переменными: 
#!/bin/bash

# Объявление переменных (БЕЗ пробелов!)
NAME="Иван"
AGE=25
PI=3.14

# Использование переменных
echo "Меня зовут $NAME"
echo "Мне $AGE лет"
echo "Значение PI: $PI"

# Рекомендуется использовать фигурные скобки
echo "Меня зовут ${NAME}"
echo "${NAME}ович"  # Объединение с текстом

# Переменные только для чтения
readonly CONSTANT="Нельзя изменить"
declare -r ANOTHER_CONSTANT="Тоже нельзя"
⚠️ Важно:
  • НЕ СТАВЬТЕ пробелы вокруг знака =
  • NAME="Иван" ✅ правильно
  • NAME = "Иван" ❌ ошибка
  • Имена переменных чувствительны к регистру: name ≠ NAME

Арифметические операции

📖 Вычисления с переменными: 
#!/bin/bash

# Метод 1: $(( ))
A=10
B=5
SUM=$((A + B))
DIFF=$((A - B))
MULT=$((A * B))
DIV=$((A / B))
MOD=$((A % B))

echo "Сумма: $SUM"
echo "Разность: $DIFF"
echo "Произведение: $MULT"
echo "Деление: $DIV"
echo "Остаток: $MOD"

# Инкремент и декремент
COUNT=0
((COUNT++))  # COUNT = COUNT + 1
echo "COUNT: $COUNT"

# Метод 2: let
let RESULT=A+B
let "RESULT = A + B"  # с пробелами нужны кавычки

# Метод 3: expr (старый способ)
RESULT=$(expr $A + $B)

# Для дробных чисел используйте bc
PI=$(echo "scale=2; 22/7" | bc)
echo "PI ≈ $PI"

🔀 Условные операторы

Оператор if

📖 Базовый синтаксис if: 
#!/bin/bash

AGE=18

# Простой if
if [ $AGE -ge 18 ]; then
    echo "Вы совершеннолетний"
fi

# if-else
if [ $AGE -ge 18 ]; then
    echo "Вы совершеннолетний"
else
    echo "Вы несовершеннолетний"
fi

# if-elif-else
if [ $AGE -lt 13 ]; then
    echo "Ребёнок"
elif [ $AGE -lt 18 ]; then
    echo "Подросток"
elif [ $AGE -lt 65 ]; then
    echo "Взрослый"
else
    echo "Пенсионер"
fi
💡 Важные моменты синтаксиса:
  • Обязательны пробелы после [ и перед ]
  • [ $AGE -ge 18 ] ✅ правильно
  • [$AGE -ge 18] ❌ ошибка
  • then может быть на той же строке с ; или на новой
  • Закрывающее fi обязательно

Операторы сравнения чисел

Оператор Описание Пример
-eq Равно (equal) [ $A -eq $B ]
-ne Не равно (not equal) [ $A -ne $B ]
-gt Больше (greater than) [ $A -gt $B ]
-ge Больше или равно [ $A -ge $B ]
-lt Меньше (less than) [ $A -lt $B ]
-le Меньше или равно [ $A -le $B ]

Операторы сравнения строк

Оператор Описание Пример
= или == Строки равны [ "$A" = "$B" ]
!= Строки не равны [ "$A" != "$B" ]
< Меньше (лексикографически) [[ "$A" < "$B" ]]
> Больше (лексикографически) [[ "$A" > "$B" ]]
-z Строка пустая [ -z "$STR" ]
-n Строка не пустая [ -n "$STR" ]
📖 Примеры сравнения строк: 
#!/bin/bash

NAME="Иван"

# Проверка равенства
if [ "$NAME" = "Иван" ]; then
    echo "Привет, Иван!"
fi

# Проверка на пустоту
if [ -z "$NAME" ]; then
    echo "Имя не задано"
else
    echo "Имя: $NAME"
fi

# Проверка на непустоту
if [ -n "$NAME" ]; then
    echo "Имя задано: $NAME"
fi

Проверка файлов и каталогов

Оператор Описание
-e Файл существует
-f Обычный файл существует
-d Каталог существует
-s Файл не пустой
-r Файл читаемый
-w Файл доступен для записи
-x Файл исполняемый
-L Файл - символическая ссылка
📖 Примеры проверки файлов: 
#!/bin/bash

FILE="test.txt"

# Проверка существования файла
if [ -f "$FILE" ]; then
    echo "Файл $FILE существует"
else
    echo "Файл $FILE не найден"
fi

# Проверка каталога
if [ -d "/home/user" ]; then
    echo "Каталог существует"
fi

# Проверка прав
if [ -r "$FILE" ]; then
    echo "Файл читаемый"
fi

if [ -w "$FILE" ]; then
    echo "Файл доступен для записи"
fi

if [ -x "script.sh" ]; then
    echo "Скрипт исполняемый"
fi

Логические операторы

📖 AND, OR, NOT: 
#!/bin/bash

AGE=25
NAME="Иван"

# AND (&&)
if [ $AGE -ge 18 ] && [ "$NAME" = "Иван" ]; then
    echo "Совершеннолетний Иван"
fi

# OR (||)
if [ $AGE -lt 18 ] || [ $AGE -gt 65 ]; then
    echo "Льготный возраст"
fi

# NOT (!)
if [ ! -f "file.txt" ]; then
    echo "Файл не существует"
fi

# Внутри [ ] используйте -a и -o
if [ $AGE -ge 18 -a "$NAME" = "Иван" ]; then
    echo "AND внутри скобок"
fi

# В [[ ]] можно использовать && и ||
if [[ $AGE -ge 18 && "$NAME" = "Иван" ]]; then
    echo "Двойные скобки поддерживают &&"
fi

Оператор case

📖 Множественный выбор: 
#!/bin/bash

        ;;
    -d|--debug)

esac

Цикл while

📖 Цикл while: 
#!/bin/bash

# Простой цикл while
COUNT=1
while [ $COUNT -le 5 ]
do
    echo "Счётчик: $COUNT"
    ((COUNT++))
done

# Бесконечный цикл
while true
do
    echo "Бесконечный цикл (Ctrl+C для выхода)"
    sleep 1
done

# Чтение файла построчно
while IFS= read -r LINE
do
    echo "Строка: $LINE"
done < file.txt

# Цикл с условием
while [ ! -f "ready.flag" ]
do
    echo "Ожидание файла ready.flag..."
    sleep 5
done
echo "Файл появился!"

Цикл until

📖 Цикл until (пока условие НЕ истинно): 
#!/bin/bash

# until - противоположность while
COUNT=1
until [ $COUNT -gt 5 ]
do
    echo "Счётчик: $COUNT"
    ((COUNT++))
done

# Ожидание появления файла
until [ -f "ready.flag" ]
do
    echo "Ожидание..."
    sleep 5
done

# Ожидание успешного выполнения команды
until ping -c 1 google.com &> /dev/null
do
    echo "Нет соединения, повтор через 5 сек..."
    sleep 5
done
echo "Соединение восстановлено!"

Управление циклами

📖 break и continue: 
#!/bin/bash

# break - выход из цикла
for i in {1..10}
do
    if [ $i -eq 5 ]; then
        echo "Прерываем цикл на 5"
        break
    fi
    echo "Число: $i"
done

# continue - пропустить текущую итерацию
for i in {1..10}
do
    if [ $((i % 2)) -eq 0 ]; then
        continue  # пропускаем чётные
    fi
    echo "Нечётное: $i"
done

# Вложенные циклы с break
for i in {1..3}
do
    for j in {1..3}
    do
        if [ $j -eq 2 ]; then
            break  # выход из внутреннего цикла
        fi
        echo "i=$i, j=$j"
    done
done

Возвращаемые значения

📖 Return и echo: 
#!/bin/bash

# return возвращает код (0-255)
is_even() {
    if [ $(($1 % 2)) -eq 0 ]; then
        return 0  # успех (истина)
    else
        return 1  # неудача (ложь)
    fi
}

if is_even 4; then
    echo "4 - чётное"
fi

# echo для возврата значений
multiply() {
    local result=$(($1 * $2))
    echo $result
}

RESULT=$(multiply 5 3)
echo "5 * 3 = $RESULT"

# Возврат нескольких значений
get_datetime() {
    echo "$(date +%Y-%m-%d)"
    echo "$(date +%H:%M:%S)"
}

# Чтение в массив
read DATE TIME <<< "$(get_datetime)"
echo "Дата: $DATE"
echo "Время: $TIME"

Рекурсивные функции

📖 Факториал (рекурсия): 
#!/bin/bash

factorial() {
    local n=$1
    
    if [ $n -le 1 ]; then
        echo 1
    else
        local prev=$(factorial $((n - 1)))
        echo $((n * prev))
    fi
}

echo "5! = $(factorial 5)"
echo "10! = $(factorial 10)"

# Числа Фибоначчи
fibonacci() {
    local n=$1
    
    if [ $n -le 1 ]; then
        echo $n
    else
        local a=$(fibonacci $((n - 1)))
        local b=$(fibonacci $((n - 2)))
        echo $((a + b))
    fi
}

echo "Fibonacci(10) = $(fibonacci 10)"

#!/bin/bash

# Значения по умолчанию
VERBOSE=false
OUTPUT=""
MODE="normal"

# Функция справки
usage() {
    cat << EOF
Использование: $0 [ОПЦИИ] файл

Опции:
    -h, --help          Показать справку
    -v, --verbose       Подробный вывод
    -o, --output FILE   Выходной файл
    -m, --mode MODE     Режим: normal|fast|safe
    
Примеры:
    $0 input.txt
    $0 -v -o output.txt input.txt
    $0 --mode fast --verbose input.txt
EOF
    exit 0
}

# Обработка аргументов
while [[ $# -gt 0 ]]; do
    case $1 in
        -h|--help)
            usage
            ;;
        -v|--verbose)
            VERBOSE=true
            shift
            ;;
        -o|--output)
            OUTPUT="$2"
            shift 2
            ;;
        -m|--mode)
            MODE="$2"
            if [[ ! "$MODE" =~ ^(normal|fast|safe)$ ]]; then
                echo "Ошибка: неверный режим '$MODE'"
                exit 1
            fi
            shift 2
            ;;
        -*)
            echo "Ошибка: неизвестная опция $1"
            usage
            ;;
        *)
            INPUT_FILE="$1"
            shift
            ;;
    esac
done

# Проверка обязательных аргументов
if [ -z "$INPUT_FILE" ]; then
    echo "Ошибка: не указан входной файл"
    usage
fi

# Использование
if [ "$VERBOSE" = true ]; then
    echo "Входной файл: $INPUT_FILE"
    echo "Выходной файл: ${OUTPUT:-не указан}"
    echo "Режим: $MODE"
fi

    echo "✓ Резервная копия создана: $BACKUP_FILE"
    SIZE=$(du -h "$BACKUP_DIR/$BACKUP_FILE" | cut -f1)
    echo "  Размер: $SIZE"
else
    echo "✗ Ошибка создания резервной копии"
    exit 1
fi

# Удаление старых бэкапов (старше 7 дней)
echo "Удаление старых копий..."
find "$BACKUP_DIR" -name "backup_*.tar.gz" -mtime +7 -delete
echo "✓ Готово"

#!/bin/bash

# Цвета
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m' # No Color

echo "=== Мониторинг системы ==="
echo

# Использование CPU
echo "CPU:"
CPU_USAGE=$(top -bn1 | grep "Cpu(s)" | awk '{print $2}' | cut -d'%' -f1)
echo "  Использование: ${CPU_USAGE}%"

# Использование памяти
echo "Память:"
MEM_TOTAL=$(free -h | awk '/^Mem:/ {print $2}')
MEM_USED=$(free -h | awk '/^Mem:/ {print $3}')
MEM_PERCENT=$(free | awk '/^Mem:/ {printf "%.1f", $3/$2 * 100}')
echo "  Всего: $MEM_TOTAL"
echo "  Использовано: $MEM_USED ($MEM_PERCENT%)"

# Использование диска
echo "Диск:"
DISK_USAGE=$(df -h / | tail -1 | awk '{print $5}' | cut -d'%' -f1)
DISK_AVAIL=$(df -h / | tail -1 | awk '{print $4}')
echo "  Использовано: ${DISK_USAGE}%"
echo "  Доступно: $DISK_AVAIL"

# Предупреждения
if [ $DISK_USAGE -gt 90 ]; then
    echo -e "${RED}⚠ ВНИМАНИЕ: Диск заполнен более чем на 90%${NC}"
fi

if (( $(echo "$MEM_PERCENT > 90" | bc -l) )); then
    echo -e "${RED}⚠ ВНИМАНИЕ: Память заполнена более чем на 90%${NC}"
fi

# Uptime
echo
echo "Uptime: $(uptime -p)"

#!/bin/bash

# Переименование всех .jpeg в .jpg

COUNT=0

for FILE in *.jpeg
do
    # Проверка существования файлов
    [ -e "$FILE" ] || continue
    
    # Новое имя
    NEWNAME="${FILE%.jpeg}.jpg"
    
    # Переименование
    mv "$FILE" "$NEWNAME"
    echo "Переименован: $FILE → $NEWNAME"
    ((COUNT++))
done

echo "Всего переименовано: $COUNT файлов"

# Вариант 2: добавление префикса
PREFIX="photo_"
for FILE in *.jpg
do
    [ -e "$FILE" ] || continue
    NEWNAME="${PREFIX}${FILE}"
    mv "$FILE" "$NEWNAME"
done

# Вариант 3: замена пробелов на подчёркивания
for FILE in *\ *
do
    [ -e "$FILE" ] || continue
    NEWNAME="${FILE// /_}"
    mv "$FILE" "$NEWNAME"
done
